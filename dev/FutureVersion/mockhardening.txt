# Find all method calls on core that we need to harden against being mocked out:
found = []
mods  = Object.constants.select { |c| Object.const_get(c).kind_of?(Module) }.sort.join('|')
calls = /(?:#{mods})\.\w+/
Dir.glob('**/*.rb') { |path|
  found.concat(File.read(path).scan(calls))
}
found.uniq
